محەممەد  کوڕی محەممەد سادقی کوڕی محەممەد مەھدی کوڕی ئیسماعیل ئەلسەدر  پیاوێکی ئایینی و مەرجەعێکی شیعە مەزھەبی عێراقی ناسراو بوو، کە دژی ڕژێمی بەعس و سەددام حوسێن چالاکی کرد.
محەممەد محەممەد سادق سەدر لە ٢٣ ئازاری ساڵی ١٩٤٣ لە ناوچەی کازمیە لەبەغدا لەبنەماڵەیەکی ئایین پەروەر  لەدایک بووە، ھاوسەر گیری لەگەڵ کچی مامی خۆیدا کردووە و چوار کوڕیان بووە بەناوی موستەفا و موقتەدا و موئەمەل و مورتەزاوە.
محەممەد  ھەر لەتەمەنی منداڵیەوە لەحەوزەی زانستی خوێندوێتی و لەتەمەنی ١١ ساڵیدا وانەی نەحوی خوێندووە، لەساڵی ١٩٥٧ چۆتە زانکۆی فیقە و دواتر درێژەی بەخوێندن داوە و لەسەر دەستی خومەینی و خۆیی وانەی خوێندووە. لە شاری نەجەف جێگیر بووە و ھەوڵی داوە پێڕەوی مەزھەبی شیعە بە ھەموو عێراقدا بڵاوبکاتەوە ھەرچەندە لەو کاتە قەدەغەبووە بەڵام لە مزگەوتی کووفە لە نەجەف لەسەر مەزھەبی شیعە نۆێژی جومعە و جەماعەتی دامەزراندووە ئەمەش بۆیە مایەیی نیگەرانی ڕژێمی سەددام حوسێن، ھەروەھا لە ساڵی ١٩٩١ وەکوو سەرکردەی ڕاپەڕینی شەعبانییە بەشداری کردووە بەڵام لەلایەن دارودەستەکانی ڕژێمی بەعسەوە دەستگیرکراوە.
محەممەد سەدر زیاتر لە ٣٠ دیوانی نووسیوە لەبواری ئاییندا و بووەتە سەرچاوەیەکی ڕوحی بۆ زۆربەی شوێنکەوتوەکانی مەزھەبی جەعفەری ھەر بۆیە ڕژێمی سەددام حسێن وەکو دوژمنی خۆی سەیری دەکرد، لە ١٩ی شوباتی ساڵی ١٩٩٩ لەگەڵ ھەردوو کوڕەکەیدا موئەمەل و موستەفا کاتێک بەئۆتۆمبێلەکەی دەگەڕانەوە بۆ ماڵی خۆیان بۆ نەجەف  ئۆتۆمبێلێکی میتسۆبۆشی خۆیپیاکێشاوە و دواتر چەند چەکدارێک لەمیتسۆبۆشێکەوە دابەزیون و خۆیی و ھەردوو کوڕەکەیان داوەتە بەر دەستڕێژی گوللە، موئەمەلی کوڕی دەستی بەجێ گیانی لەدەستداوە بەڵام خۆی گەیندراوەتە نەخۆشخانە و لەوێ لەلایەن کەسێک بە ناوی حەمزە زوبەیدی فیشەکێکی ناوە بەسەرێوە و کوشتوێتی، ھەروەھا موستەفای کوڕیشی بەھۆی سەختی برینەکەیەوە گیانی لەدەستداوە.
